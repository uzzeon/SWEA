# [SWEA] 2070번. 큰 놈, 작은 놈, 같은 놈



```python
T = int(input())
# 여러개의 테스트 케이스가 주어지므로, 각각을 처리합니다.
for test_case in range(1, T + 1):
    numbers = list(map(int, input().split()))

    if numbers[0] > numbers[1]:
        answer = ">"
    elif numbers[0] < numbers[1]:
        answer = "<"
    else:
        answer = "="
        
    print(f'#{test_case} {answer}') 
```
